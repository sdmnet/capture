⚙️ إعدادات الخادم
ربط التطبيق بقاعدة بيانات PostgreSQL
ℹ️ يتطلب عمل التطبيق خادم API وسيط يستقبل البيانات ويحفظها في PostgreSQL. قم بإعداد الخادم باستخدام Node.js/Express أو أي إطار عمل آخر.
مخطط قاعدة البيانات
🗄️
جدول public.capture
ID
integer [PK] — auto
Photo
bytea — بيانات الصورة
longitude
double precision
Latitude
double precision
Data
date — تاريخ الحادث
time
time with time zone
كود الخادم (Node.js)
const express = require('express'); const { Pool } = require('pg'); const multer = require('multer'); const app = express(); const pool = new Pool({ connectionString: process.env.DB_URL }); const upload = multer({ storage: multer.memoryStorage() }); app.use((req,res,next)=>{ res.header('Access-Control-Allow-Origin','*'); next(); }); app.post('/api/capture', upload.single('photo'), async (req, res) => { const { longitude, latitude, date, time } = req.body; const photo = req.file?.buffer; await pool.query( 'INSERT INTO public.capture("Photo","longitude","Latitude","Data","time") VALUES($1,$2,$3,$4,$5)', [photo, longitude, latitude, date, time] ); res.json({ success: true }); }); app.listen(3000);
